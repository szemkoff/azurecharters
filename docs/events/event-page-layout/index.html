Analytics Integration​

// Event tracking
function trackEventView(eventId, eventName) {
    analytics.track('Event View', {
        eventId: eventId,
        eventName: eventName,
        timestamp: new Date()
    });
}

// Booking funnel tracking
function trackBookingStep(step, eventType) {
    analytics.track('Booking Step', {
        step: step,
        eventType: eventType,
        timestamp: new Date()
    });
}
